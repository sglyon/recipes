% Italian Chicken Noodle Soup
\begin{recipe}
    [preparationtime,
     bakingtime,
     bakingtemperature,
     portion,
     calory,
     source
    ]
    % Put the name of the recipe in between the { } on the next line
    {Italian Chicken Noodle Soup}
    \index{chicken}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients{
            1 Tbsp. & oil \\
            2 & bone/skin-less chicken brst\\
            1 medium & onion, chopped \\
            2 cans & chicken broth \\
            2 c. & water \\
            3 medium & carrots, sliced \\
            2 c. & broccoli florets \\
            1 1/2 c. & dry medium egg noodles \\
            1 tsp. & dried basil leaves \\
            1/2 tsp. & garlic-pepper blend \\
            1/4 c. & shredded parmesan cheese
        }


        \preparation{
            \step Cut the chicken into bite-sized cubes.
            \step In 4-quart saucepan, heat oil over medium heat.  Add chicken.  Cook 4-6 minutes, stirring occasionally until no longer pink in center.
            \step Stir in onion.  Cook 2-3 minutes, stirring occasionally until onion is tender.
            \step Stir in broth, water, and carrots.  Heat to boiling.  Cook 5 minutes over medium heat.
            \step Stir in broccoli, noodles, basil, ad garlic-pepper blend.  Heat to boiling then reduce heat.
            \step Simmer uncovered for 8-10 minutes, stirring occasionally, until vegetables and noodles are tender.
            \step Top each serving with cheese.
        }
\end{recipe}
