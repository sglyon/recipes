% Potato Soup
\begin{recipe}
    [preparationtime,
     bakingtime,
     bakingtemperature,
     portion,
     calory,
     source = Michelle Hollis
    ]
    % Put the name of the recipe in between the { } on the next line
    {Potato Soup}
    \index{potato}

        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {    5 Large & Potatoes\\
            2/3 c. & butter\\
            2/3 c. &  flour\\
            8 c. & milk\\
            4 & green onions, sliced\\
            2 c. & grated cheese\\
            8 oz. & sour cream \\
            1 lb. & bacon (fried and crumbled)\\
            1 tsp. & salt\\
            1/2 tsp. & pepper\\
            1/2 tsp. & seasoned salt
        }

        \preparation{
            \step Bake potatoes at 375 degrees for 1-1 1/2 hours.  Peel and cool.
            \step In a large pan, melt butter.  Whisk in flour with butter until smooth.
            \step Add milk, potatoes and onions.
            \step Stir till almost boiling.
            \step Turn down heat and simmer for 10 minutes then add remaining ingredients.
        }
\end{recipe}
