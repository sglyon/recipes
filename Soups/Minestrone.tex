% Minestrone
\begin{recipe}
    [preparationtime,
     bakingtime,
     bakingtemperature,
     portion,
     calory,
     source = Betty Crocker
    ]
    % Put the name of the recipe in between the { } on the next line
    {Minestrone}
    \index{vegetarian}
    \index{sausage}

        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {    1 Tbsp. &  oil \\
            1 lb. &  italian sausage \\
            1 medium  & onion, chopped \\
            2 medium  & carrots, coarsely chopped \\
            2 tsp. &  dried basil leaves \\
            2 tsp. &  finely chopped garlic \\
            3 cans  & beef broth \\
            1 can  & diced tomatoes \\
            1 can  & great northern beans \\
            1 c. &  uncooked small elbow macaroni \\
            1 medium  & zucchini\\
            1 c. &  frozen cut green beans
        }


        \preparation{
            \step Prepare ingredients by rinsing and draining the great northern beans and cutting the zucchini lengthwise in half, then cut into 1/4 inch slices.
            \step In 5-quart pan, heat oil over medium-high heat.  Cook sausage, onion, carrots, basil and garlic in oil 5-7 minutes stirring frequently, until sausage is no longer pink; drain grease.
            \step Stir broth tomatoes and great northern beans into sausage mixture.
            \step Heat to boiling then reduce heat to medium-low.
            \step  Cover and cook 7-8 minutes, stirring occasionally. Stir in macaroni, zucchini, and green beans.
            \step Heat to boiling.
            \step Cook over medium-high heat 5-6 minutes, stirring occasionally, until vegetables are hot ad macaroni is tender.
        }
        \hint{Make it meatless by substituting an additional can of great northern beans for sausage and use vegetable broth instead of beef broth.}
\end{recipe}
