% Snickerdoodles
\begin{recipe}
    [preparationtime,
     bakingtime= {8 min},
     bakingtemperature = {375 \degf},
     portion,
     calory,
     source = Kelli Skousen
    ]
    {Snickerdoodles}
    \index{cinnamon}
    \index{cookie}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        3 3/4 c. & flour\\
        1 tsp. & baking soda\\
        1/2 tsp. & cream of tarter\\
        1/2 tsp. & salt\\
        1 c. & butter\\
        2 c. & sugar\\
        2 & eggs\\
        1/4 c. & milk\\
        1 tsp. & vanilla
        }
        \preparation{
            \step Mix dry ingredients in small bow.
            \step In seperate, large bowl, cream butter and sugar till fluffy.
            \step  Add eggs, milk, and vanilla.
            \step Slowly mix in dry ingredients.
            \step Roll small cookie dough balls in mix of 4 Tbsp. sugar and 1 tsp. cinnamon.
            \step Bake at 375 degrees for 8 minutes.
        }
\end{recipe}
