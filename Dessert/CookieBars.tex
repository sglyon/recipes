% Cookie Bars
\begin{recipe}
    [preparationtime,
     bakingtime= {25min},
     bakingtemperature = {375 \degf},
     portion,
     calory,
     source = Julie Bradshaw
    ]
    {Cookie Bars}
    \index{butterscotch}
    \index{chocolate}
    \index{cookie}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1 pkg. & chocolate or butterscotch chips\\
        1/4 tsp. & baking soda\\
        2/3 c. & margarine\\
        1 c. & sugar\\
        1 tsp. & vanilla\\
        1 tsp. & baking powder\\
        1 tsp. & salt\\
        1 c. & brown sugar\\
        2 c. & flour\\
        2 & eggs
        }
        \preparation{
            \step Melt margarine on stove top and mix with sugars and vanilla.
            \step  Remove from heat.
            \step Add eggs, stir.
            \step  Add salt, stir.
            \step  Add remaining ingredients.
            \step Pour into buttered 9x13 pan.
            \step  Bake at 375 degrees for one minute.
            \step  Take out and pour in chocolate chips and marble (mix with utensil).
            \step Bake for 25 minutes.
        }
\end{recipe}
