% Vanilla Sheet Cake
\begin{recipe}
    [preparationtime,
     bakingtime= {8 min},
     bakingtemperature = {375 \degf},
     portion,
     calory,
     source = Stephanie Skousen
    ]
    {Vanilla Sheet Cake}
    \index{cake}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1/2 c. &  buttermilk\\
        2 & eggs\\
        2 tsp. &  vanilla\\
        2 c. &  sugar\\
        2 c. &  flour\\
        2 sticks & margarine\\
        3/4 c. &  water\\
        1 tsp. &  baking soda\\
        1 box & powdered sugar\\
        1/3 c. &  milk\\
        1/4 c. &  shortening
        }
        \preparation{
            \step Melt 1 stick margarine, shortening, and water.
            \step Add sugar, flour and mix.
            \step  Add buttermilk, eggs, 1 tsp. vanilla and mix.
            \step  MIx in baking soda.
            \step  Pour into greased jelly-roll pan.
            \step  Bake at 400 degrees for 15-20 minutes.
            \step  5 minutes before cake is done, melt margarine, milk. Bring to a boil, stirring constantly.  Add powdered sugar and 1 tsp. vanilla.
            \step Pour over cake.
        }
\end{recipe}
