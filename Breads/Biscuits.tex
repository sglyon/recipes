% Biscuits
\begin{recipe}
    [preparationtime,
     bakingtime= {15-20 mins.},
     bakingtemperature = {400 \degf},
     portion,
     calory,
     source = Becky Anderson
    ]
    {Biscuits}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        2 c. & flour \\
        2 Tbsp. & sugar \\
        1/2 c. & shortening \\
        2 1/2 tsp. & baking powder \\
        3/4 tsp. & salt
        }

        \preparation{
            \step Put all ingredients in bowl and Cut with two knives.
            \step Add one well-beaten egg, 3/4 c. milk and stir till moist.
            \step Place in well-greased tin.
            \step Sprinkle with cinnamon-sugar.
            \step Bake at 400 degrees for 15-20 minutes.
        }
\end{recipe}
