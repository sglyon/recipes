% Pizza Dough
\begin{recipe}
    [preparationtime,
     bakingtime= {10-12 mins.},
     bakingtemperature = {400 \degf},
     portion,
     calory,
     source = Dianne Hathcock
    ]
    {Pizza Dough}
    \index{pizza}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1 Tbsp. & yeast \\
        1 1/2 tsp. & sugar\\
        3/4 c. & warm water\\
        1/2 tsp. & salt\\
        2 c. & flour
        }

        \preparation{
            \step Pour 1/2 c. warm water into mixing bowl.
            \step Add yeast and sugar.  Stir to dissolve.
            \step Add 1/4 c. water, flour and salt.
            \step Let raise for 30 minutes.
            \step Bake at 400 degrees for 10-12 minutes.
        }
\end{recipe}
