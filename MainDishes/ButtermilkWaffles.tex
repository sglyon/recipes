% Buttermilk Waffles
\begin{recipe}
    [preparationtime,
     bakingtime,
     bakingtemperature,
     portion,
     calory,
     source = Jen McIff
    ]
    {Buttermilk Waffles}
     \index{breakfast}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1 1/2 c. & flour\\
        1 tsp. & scant baking soda\\
        1 tsp. & scant salt\\
        3 tsp. & baking powder\\
        3 & eggs, beaten\\
        1 1/2 c. & buttermilk\\
        1/2 c. & melted butter.
        }
        \preparation{
            \step Mix dry ingredients.
            \step Alternate mixing eggs and buttermilk into dry ingredients.
            \step Add butter.
            \step Cook like normal waffles.
        }
\end{recipe}
