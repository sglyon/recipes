% Hawaiian Roll Sandwiches
\begin{recipe}
    [preparationtime,
     bakingtime= {15-20 min},
     bakingtemperature={350 \degf},
     portion,
     calory,
     source = Kelli Skousen
    ]
    {Hawaiian Roll Sandwiches}
    \index{sandwich}
    \index{quick}
    \index{basic}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1/2 c. & melted butter \\
        1 tsp. & garlic powder \\
        3 Tbsp. & Warstershire sauce \\
        1 Tbsp. & poppyseeds \\
        12 & Hawaiian Rolls \\
        12 & servings sandwich meat \\
        12 & servings sandwich cheese
        }

        \preparation{
            \step Preheat oven to 350 degrees.
            \step Place sandwiches in 9x13 pan.  Pour sauce over top.
            \step Bake 15-20 minutes.
            \step Remove from pan and let sit for a few (5ish) minutes.
        }
      \hint{Best with ham and provolone cheese}
\end{recipe}
