% Stroganoff
\begin{recipe}
    [preparationtime,
     bakingtime= {2-4 hr},
     bakingtemperature,
     portion,
     calory,
     source = Kelli Skousen
    ]
    {Stroganoff}
    \index{beef}
    \index{basic}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1 package & of stew meat \\
        2 cans & cream of mushroom soup \\
        1 can & cream of celery soup \\
        1 package & of onion soup mix \\
        1/2 c. & sour cream
        }

        \preparation{
            \step Brown stew meat in olive oil.
            \step Put in crock pot.  Add soups and onion mix.
            \step When finished cooking add sour cream.
        }
\end{recipe}
