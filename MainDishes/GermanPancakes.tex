% German Pancakes
\begin{recipe}
    [preparationtime,
     bakingtime= {20 min},
     bakingtemperature = {400 \degf},
     portion,
     calory,
     source = Kelli Skousen
    ]
    {German Pancakes}
    \index{breakfast}
    \index{eggs}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        6 & eggs\\
        1 c. & flour\\
        1 c. & milk\\
        1/4 tsp. & salt
        }
        \preparation{
            \step Blend ingredients.
            \step Pre-heat oven to 400 degrees.
            \step Melt one stick of butter in 9x13 pan.
            \step  Pour batter over melted butter.
            \step Bake for 20 minutes.
        }
\end{recipe}
