% Mexican Lasagna
\begin{recipe}
    [preparationtime,
     bakingtime= {30 min},
     bakingtemperature = {350 \degf},
     portion,
     calory,
     source = Betty Crocker
    ]
    {Mexican Lasagna}
    \index{mexican}
    \index{chicken}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        12 & uncooked  lasagna noodles\\
        2 can & cream chicken soup\\
        1 8 oz & container of sour cream\\
        1/4 c. & milk\\
        1 1/4 tsp. & ground cumin\\
        1/2 tsp. & garlic powder\\
        3 c. & cubed, cooked chicken (see tip)\\
        1 can & chopped green chillies\\
        8-10 med. & green onions, sliced\\
        1/2 c. & chopped fresh cilantro or parsley\\
        3 c. & finely shredded  cheese blend\\
        1 lg. & red bell pepper, chopped\\
        1 c. & tortilla chips
        }
        \preparation{
            \step Heat oven to 350 degrees.
            \step Cook and drain noodles as directed on package.
            \step Meanwhile, in large bowl, mix soup, sour cream, milk, cumin, garlic powder, chicken and chiles.
            \step Spread about 1 1/4 cups of chicken mixture in baking dish.  Top with 4 noodles.
            \step Spread 1 1/4 c. chicken mixture over noodles; sprinkle with onions and cilantro.  Sprinkle with 1 c. cheese.  Top with 4 noodles.
            \step Spread 1/4 c. chicken mixture over noodles; sprinkle with bell pepper.  Sprinkle with 1 c. of cheese.  Top with 4 noodles; spread with remaining chicken mixture.
            \step Crush chips on top and sprinkle with remaining cheese.
            \step Cook for 30 minutes.
        }
      \hint{1 can of cubed chicken is perfect. Also, Tostitos tortilla chips with a "hint of lime" are very good}
\end{recipe}
