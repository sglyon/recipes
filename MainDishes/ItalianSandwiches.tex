% Italian Sandwiches
\begin{recipe}
    [preparationtime,
     bakingtime= {2-4 hr},
     bakingtemperature,
     portion,
     calory,
     source = Shannen Lyon
    ]
    {Italian Sandwiches}
    \index{sandwich}
    \index{sausage}
    \index{beef}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1/2 lb. & sausage \\
        1/2 lb. & Ground beef or turkey \\
        1 small & onion, chopped \\
        1 & green bell pepper, chopped \\
        1 & red bell pepper, chopped
        }

        \preparation{
            \step Brown meat.  and flavor it with a small amount of garlic and dried cilantro leaves.
            \step Place meat in crock-pot, layering with onions and peppers.
            \step Cook on low for 2-4 hours.
            \step Just prior to serving, broil sandwich rolls till golden brown.
            \step Add meat mixture and slice of favorite cheese;  Broil till cheese is melted.
        }
      \hint{Provolone cheese is very good.}
\end{recipe}
