% Lasagna
\begin{recipe}
    [preparationtime,
     bakingtime= {45-60 min},
     bakingtemperature={375 \degf},
     portion,
     calory,
     source = Cindy Himes
    ]
    {Lasagna}
    \index{italian}
    \index{sausage}
        \graph
        {% pictures I will help you figure out how to put them in here. Just ask me.
            small,
            big
        }

        \ingredients
        {
        1 lb. & sausage\\
        1 Jar & spaghetti sauce\\
        1 c. & cottage cheese\\
        1/2 c. & grated parmesan \\
        2 & eggs\\
        1 pkg. & lasagna noodles\\
        & Cheddar and Mozzarella cheese
        }

        \preparation{
            \step Brown meat.
            \step Pour sauce in meat and let simmer.
            \step In small mixing bowl mix cottage cheese, eggs, and parmesan cheese.
            \step Pour small amount of sauce in 9x13 pan.
            \step Layer noodles, meat, cottage cheese mix, grated cheese mix.
            \step Bake at 375 degrees for 45 minutes-1 hour.
        }
\end{recipe}
